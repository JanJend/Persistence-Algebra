660  static int compare_three_way(const GradedSparseMatrix<D, index>& a, const GradedSparseMatrix<D, index>& b) {
661  // Compare row degrees

663  if (Degree_traits<D>::smaller(a.row_degrees[i], b.row_degrees[i])) {
664  return -1;
665  }
666  if (Degree_traits<D>::smaller(b.row_degrees[i], a.row_degrees[i])) {
667  return 1;
668  }
669  }
670  if (a.row_degrees.size() != b.row_degrees.size()) {
671  return a.row_degrees.size() < b.row_degrees.size() ? -1 : 1;
672  }
673 
674  // Compare column degrees
675  for (size_t i = 0; i < std::min(a.col_degrees.size(), b.col_degrees.size()); ++i) {
676  if (Degree_traits<D>::smaller(a.col_degrees[i], b.col_degrees[i])) {
677  return -1;
678  }
679  if (Degree_traits<D>::smaller(b.col_degrees[i], a.col_degrees[i])) {
680  return 1;
681  }
682  }
683  if (a.col_degrees.size() != b.col_degrees.size()) {
684  return a.col_degrees.size() < b.col_degrees.size() ? -1 : 1;
685  }
686 
687  return 0;
688  }

788 template <typename D, typename index>
789 std::pair< SparseMatrix<index>, vec<std::pair<index, index> > > block_hom_space_without_optimisation(const GradedSparseMatrix<D, index>& A, const GradedSparseMatrix<D, index>& C, const GradedSparseMatrix<D, index>& B,
790  vec<index>& C_rows, vec<index>& B_rows, bool system_size = false) {
791  vec<std::pair<index, index>> row_ops; // we store the matrices Q_i which form the basis of hom(C, B) as vectors
792  // This translates from entries of the vector to entries of the matrix.
793  SparseMatrix K(0,0);
794  SparseMatrix S(0,0);
795  S.data.reserve( C_rows.size() + B_rows.size() + 1);
796  index S_index = 0;
797  // First add all row-operations from C to B
798  for(index i = 0; i < C_rows.size(); i++){
799  for(index j = 0; j < B_rows.size(); j++){
800  auto source_row_index = C_rows[i];
801  auto target_row_index = B_rows[j];
802  if(A.is_admissible_row_operation(source_row_index, target_row_index)){
803  row_ops.push_back({source_row_index, target_row_index});
804  S.data.push_back(vec<index>());
805  for(auto rit = C._rows[i].rbegin(); rit != C._rows[i].rend(); rit++){
806  auto& column_index = *rit;
807  S.data[S_index].emplace_back(A.linearise_position_reverse(column_index, target_row_index));
808  }
809  S_index++;
810  }
811  }
812  }
813 
814  index row_op_threshold = S_index;
815  assert( row_ops.size() == S_index );
816 
817  if(row_op_threshold == 0){
818  // If there are no row-operations, then the hom-space is zero.
819  return {SparseMatrix<index>(), row_ops};
820  }
821 
822  // Then all column-operations from B to C
823  for(index i = 0; i < B.columns.size(); i++){
824  for(index j = 0; j < C.columns.size(); j++){
825  if(A.is_admissible_column_operation(B.columns[i], C.columns[j])){
826  S.data.push_back(vec<index>());
827  for(index row_index : B.data[i]){
828  S.data[S_index].emplace_back(A.linearise_position_reverse(C.columns[j], row_index));
829  }
830  S_index++;
831  }
832  }
833  }
834 
835  S.compute_num_cols();
836 
837  if(system_size){
838  std::cout << "System size: " << S.num_cols << std::endl;
839  }
840 
841  // If M, N present the modules, then the following computes Hom(M,N), i.e. pairs of matrices st. QM = NP.
842  K = S.get_kernel();
843  // To see how much the following reduces K: index K_size = K.data.size();
844  // Now we need to delete the entries of K which correspond to the row-operations.
845  K.cull_columns(row_op_threshold, false);
846 
847  // Last we need to quotient out those Q where for every i the column Q_i - with its degree alpha_i -
848  // lies in the image of N, that is, it lies in the image of N|alpha_i.
849  // That is equivalent to locally reducing every column of Q.
850 
851  for(index i = 0; i < C_rows.size(); i++){
852  D alpha = C.row_degrees[i];
853  vec<index> local_admissible_columns;
854  auto B_alpha = B.map_at_degree(alpha, local_admissible_columns);
855  std::unordered_map<index, index> shiftIndicesMap;
856  for(index j : B_rows){
857  shiftIndicesMap[j] = A.linearise_position_reverse(C_rows[i], j);
858  }
859  B_alpha.transform_data(shiftIndicesMap);
860  B_alpha.reduce_fully(K);
861  }
862 
863  // delete possible linear dependencies.
864  K.column_reduction_triangular(true);
865  return std::make_pair(K, row_ops);
866 }

881 template <typename index>
882 vec<index> hom_quotient( const Hom_space_temp<index>& full_space,
883  SparseMatrix<index>& coKer_B, const vec<index>& basislift_C, vec<index>& C_admissible_rows, vec<index>& B_admissible_rows, vec<index>& C_rows){
884 
885  // For each map in full_space, we need to compute the induced map at alpha.
886  // It is given by first restricting the domain to the rows in basislift_C,
887  // then composing from the left with the coKer_B map,
888 
889  // Because the basislift is given as a subset of the admissible rows,
890  // we need to re-index the map Q wth the help of the admissible_rows.
891 
892  vec<SparseMatrix<index>> induced_maps = vec<SparseMatrix<index>>();
893 
894  auto& B_row_index_map = shiftIndicesMap(B_admissible_rows);
895 
896  for(index i = 0; i < full_space.first.num_cols; i++){
897  const vec<index>& current_map = full_space.first.data[i];
898  auto& index_pairs = full_space.second; // This is a pair (i,j) where i is the column index, j, the row index, and it is sorted first by i, then by j.
899  SparseMatrix<index> Q_basislift_C = SparseMatrix<index>(0,0);
900  index k = 0;
901  for(index j : basislift_C){
902  Q_basislift_C.data.push_back(vec<index>());
903 
904  if(k != current_map.size()){
905 
906  while( C_rows[ index_pairs[current_map[k]].first ] < C_admissible_rows[j] ){
907  k++;
908  if(k == current_map.size()){
909  break;
910  }
911  }
912 
913  while( C_rows[ index_pairs[current_map[k]].first ] == C_admissible_rows[j]){
914  Q_basislift_C.data.back().push_back( B_row_index_map[ index_pairs[current_map[k]].second ]);
915  k++;
916  if(k == current_map.size()){
917  break;
918  }
919  }
920  }
921  }
922  Q_basislift_C.compute_num_cols();
923  // Now we have the restriction to the basislift_C, we need to compose with the coKer_B map.
924  //TO-DO: Compute the rows of coKer_B beforehand and then multiply to avoid transposition.
925  induced_maps.emplace_back(multiply_transpose(coKer_B, Q_basislift_C));
926  }
927 
928  // Reduce the vector space of induced homomorphisms
929  return general_reduction<index, SparseMatrix<index>> (induced_maps);
