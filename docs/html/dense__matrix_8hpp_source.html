317  void colReduce(bool complete = false) {
318  int lead = 0;
319  pivot_vector = boost::dynamic_bitset<>(num_rows, 0);
320  for (int r = 0; r < num_cols; ++r) {
321  if (lead >= num_rows) {
322  break; // No more columns to work on, exit the loop
323  }
324 
325  int i = r;
326  // Find the first non-zero entry in the row(potential pivot)
327  while (i < num_cols && !data[i][lead]) {
328  ++i;
329  }
330 
331  if (i < num_cols) {
332  // Found a non-zero entry, so this row does have a pivot
333  // If the pivot is not in the current col, swap the cols
334  if (i != r) {
335  swap_cols(i, r);
336  }
337  pivot_vector[lead] = true; // Mark this row as having a pivot after confirming pivot
338 
339  // Eliminate all non-zero entries below this pivot
340  for (int j = r + 1; j < num_cols; ++j) {
341  if (data[j][lead]) {
342  data[j] ^= data[r];
343  }
344  }
345 
346  if (complete) {
347  // Eliminate all non-zero entries above this pivot
348  for (int j = 0; j < r; ++j) {
349  if (data[j][lead]) {
350  data[j] ^= data[r];
351  }
352  }
353  }
354 
355  ++lead; // Move to the next row
356  } else {
357  // No pivot in this row, so we move to the next row without incrementing the col int
358  ++lead;
359  --r; // Stay on the same col for the next iteration
360  }
361  }
362  rowReduced = true;
363  if(complete){completeRowReduced = true;}
364  }
